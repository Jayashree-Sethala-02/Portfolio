Drift Detection SQL
// runs against Gold layer
-- Daily drift detection query WITH drift_calc AS ( SELECT g.account_id, a.client_name, g.platform, g.asset_class, ROUND(g.allocation_pct,2) current_pct, ROUND(t.target_pct,2) target_pct, ROUND(g.allocation_pct - t.target_pct,2) drift_pct, ABS(g.allocation_pct - t.target_pct) abs_drift FROM gold_portfolio g JOIN target_allocation t ON g.account_id=t.account_id AND g.asset_class=t.asset_class JOIN account_info a ON g.account_id=a.account_id ) SELECT *, CASE WHEN abs_drift>10 THEN 'URGENT' WHEN abs_drift>5 THEN 'REBALANCE' ELSE 'ON TARGET' END AS status FROM drift_calc ORDER BY abs_drift DESC;
